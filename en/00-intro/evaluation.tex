\section{Assessment}

\begin{frame}[t]{Assessment System}
\begin{itemize}
  \item In \textmark{summary}:

    \mode<presentation>{\vfill}
    \begin{itemize}
      \item \textgood{Final exam}: 40\% of final score.
        \begin{itemize}
          \item All contents included (theoretical, practical, and projects).
        \end{itemize}

      \mode<presentation>{\vfill}
      \item \textgood{Continuous evaluation}: 60\% of final score.
        \begin{itemize}
          \item \textmark{Partial exams}: 24\% of final score.
          \item \textmark{Lab exams}: 12\% of final score
          \item \textmark{Project}: 24\% of final score.
        \end{itemize}

    \mode<presentation>{\vfill}
    \item \textemph{Evaluation calls}:
      \begin{itemize}
        \item \textmark{Ordinary}: January.
        \item \textmark{Extraordinary}: June.
      \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Continuous evaluation}
\begin{itemize}
  \item Good continuous evaluation scores are \textemph{key} for success.

  \mode<presentation>{\vfill}
  \item \textgood{Elements}:
    \begin{itemize}
      \item \textmark{Partial exams}: 24\% of final score.
        \begin{itemize}
          \item 2 exams $\Rightarrow$ 12\% each one.
        \end{itemize}
      \item \textmark{Lab exams}: 12\% of final score.
        \begin{itemize}
          \item Lab exam 1: 4\%.
          \item Lab exam 2: 8\%.
        \end{itemize}
      \item \textmark{Project}: 24\% of final score.
    \end{itemize}
  \vspace{1em}

  \mode<presentation>{\vfill}
  \item You \textbad{have not followed} continuous evaluation:
    \begin{itemize}
      \item if you get less than 2.0 in any lab exam.
      \item if you get less than 2.0 in the project.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Ordinary round: Continuous evaluation}
\begin{itemize}
  \item If you follow the \textmark{continuous evaluation} process:
    \begin{itemize}
      \item \textgood{Final exam}: 40\%.
        \begin{itemize}
          \item Minimum required: \textbad{3.5}.
        \end{itemize}
      \item \textgood{Partial exams}: 24\%
        \begin{itemize}
          \item Minimum required: \textbad{There is no minimum}.
        \end{itemize}
      \item \textgood{Labs}: 12\%.
        \begin{itemize}
          \item Minimum required: \textbad{2.0} in each exam.
        \end{itemize}
      \item \textgood{Project}: 24\%.
        \begin{itemize}
          \item Minimum required: \textbad{2.0}.
        \end{itemize}
    \end{itemize}

    \mode<presentation>{\vfill}
    \item \textbad{IMPORTANT}: If you do not achieve a minimum, the average is not computed and 
        \textbad{you have failed}.
\end{itemize}
\end{frame}

\begin{frame}[t]{Improve your grades}
\begin{itemize}
  \item \textemph{Bonus}:
    \begin{itemize}
      \item 1 point is added to your score if:
        \begin{itemize}
          \item You got at least 7.0 points in continuous evaluation, and,
          \item You got at least 6.0 points in the final exam.
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Use of generative AI}
\begin{itemize}

\item The use of Generative AI is selectively allowed in the evaluable activities of the course.

\item The use of Generative AI is allowed in certain tasks in this subject:

  \begin{itemize}
    \item Any generative AI including (but not limited to) coding assistants such as co-pilots are allowed in the course project, provided that the specific used tools are declared in the project report. More information will be provided in the course project statement.
  \end{itemize}

\item It is the student's responsibility to be aware of these specific permissions.
\end{itemize}
\end{frame}

\begin{frame}[t]{Projects and conflict resolution}
\begin{itemize}
  \item \textemph{Rules}:
    \begin{itemize}
      \item Project is designed for a team of 4 people.
      \item Smaller groups are allowed but there will be no workload reduction.
      \item Individual contributions must be declared in the report.
        \begin{itemize}
          \item A contribution shall not have more than one author.
        \end{itemize}
    \end{itemize}

  \mode<presentation>{\vfill\pause}
  \item \textbad{Conflicts}:
    \begin{itemize}
      \item If there is a team conflict, report to the coordinator soon.
      \item If a conflict cannot be solved, the coordinator may:
        \begin{itemize}
          \item Remove a member from the team.
          \item Split the team in two smaller teams.
          \item No workload reduction will be authorized.
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Ordinary round: NON-Continuous evaluation}
\begin{itemize}
  \item If you \textbad{did not follow} the \textgood{continuous evaluation} process:
    \begin{itemize}

      \item You need \textbad{8.33} in the \textgood{final exam} to pass the course.
    \end{itemize}

  \mode<presentation>{\vfill}
  \item \textbad{ADVICE}:
    \begin{itemize}
      \item Try hard to follow the continuous evaluation process.
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[t]{Extraordinary round}
\begin{itemize}
  \item \textgood{Extraordinary exam} to be held in June.

  \mode<presentation>{\vfill}
  \item \textmark{Rules}:
    \begin{enumerate}
      \item The student \textgood{completed} the continuous evaluation:
        \begin{itemize}
          \item Extraordinary exam gives \textbad{40\%} 
                and continuous evaluation gives \textbad{60\%}.
          \item \textbad{Only applied} if the score in exam is at least \textbad{3.5}.
        \end{itemize}
      \item The student \textbad{did not complete} the continuous evaluation: 
        \begin{itemize}
          \item Final exam gives \textbad{100\%}.
        \end{itemize}
    \end{enumerate}

    \mode<presentation>{\vfill}
    \begin{itemize}
      \item For students having completed continuous evaluation 
            the \textgood{best of two options} is taken.
    \end{itemize}
\end{itemize}
\end{frame}
